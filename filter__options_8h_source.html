673  static_cast<id_t>(ALL_IDS) : info.flowcell().surface_count()+1;

675  static_cast<id_t>(ALL_IDS) : 1;

683  const id_t swath_end = !supports_swath(plot_type) ? static_cast<id_t>(ALL_IDS) :
